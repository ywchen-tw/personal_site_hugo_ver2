Plot Taiwan Map with matplotlib

%matplotlib inline
import matplotlib.pyplot as plt
from mpl_toolkits.basemap import Basemap
import numpy as np

# set map region
minLat, maxLat = 21.75, 25.5
minLon, maxLon = 119.25, 122.5

# White Background       
fig = plt.figure(figsize=(6, 9))
ax = fig.add_axes([0.025, 0.025, 0.925, 0.925])

# plot map
m = Basemap(projection='cyl',resolution='h',llcrnrlat=minLat, \
    urcrnrlat=maxLat, llcrnrlon=minLon, urcrnrlon=maxLon, ax=ax)
m.drawcoastlines(color='k')
m.drawcountries(color='k')
# draw parallels.
parallels = np.arange(21., 26.1, 1.)
m.drawparallels(parallels, labels=[1, 0, 0, 0], fontsize=14, linewidth=0.5)
# draw meridians
meridians = np.arange(119., 123.1, 1.)
m.drawmeridians(meridians, labels=[0, 0, 0, 1], fontsize=14, linewidth=0.5)
plt.show()

# draw meridians
meridians = np.arange(119., 123.1, 1.)
m.drawmeridians(meridians, labels=[0, 0, 0, 1], fontsize=14, linewidth=0.5, zorder=2)
# plot countries boundary
m.readshapefile('TWN_shapefile/gadm36_TWN_0', name = 'Taiwan', linewidth=0.75, color='k')
m.readshapefile('TWN_shapefile/gadm36_TWN_1', name = 'Taiwan', linewidth=0.75, color='k')
m.readshapefile('TWN_shapefile/gadm36_TWN_2', name = 'Taiwan', linewidth=0.75, color='k')
plt.show()

The shapefile used in line 16-18 can be downloaded from http://www.gadm.org/country. Choose country as “Taiwan” and then download the shapefile.
